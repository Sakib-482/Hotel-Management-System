#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define MAX_ROOMS 50

// ---------------- Data Structures ----------------
typedef struct {
    int id;
    char type[30];
    int isBooked;
} Room;

// Linked List for Booked Customers
struct Booking {
    int roomId;
    char customer[50];
    struct Booking* next;
};

// Queue for Waiting List
struct Node {
    char name[50];
    char roomType[30];
    struct Node* next;
};

// Stack for Recently Checked-out Customers
struct StackNode {
    int roomId;
    char customer[50];
    struct StackNode* next;
};

// ---------------- Global Variables ----------------
Room hotel[MAX_ROOMS];
int roomCount = 0;

struct Booking* bookingHead = NULL;
struct Node* front = NULL;
struct Node* rear = NULL;
struct StackNode* top = NULL;

// ---------------- Authentication ----------------
void registerUser() {
    char username[50], password[50];
  FILE *fp = fopen("C:\\Users\\Public\\users.txt", "a");
// append mode creates file if not exists
    if (!fp) {
        printf("Error: Unable to open or create users.txt!\n");
        return; // Exit if file can't be opened
    }
   
    printf("Enter new username: ");
    scanf("%s", username);
    printf("Enter password: ");
    scanf("%s", password);
    fprintf(fp, "%s %s\n", username, password);
    fclose(fp);
    printf(" Registration successful!\n");
}

int loginUser() {
    char username[50], password[50], fileUser[50], filePass[50];
    FILE *fp = fopen("users.txt", "r");

    printf("Enter username: ");
    scanf("%s", username);
    printf("Enter password: ");
    scanf("%s", password);

    // Default admin login
    if(strcmp(username, "admin") == 0 && strcmp(password, "DIU321") == 0) {
        printf(" Admin login successful!\n");
        return 1;
    }

    if (!fp) {
        printf("No registered users. Please register first.\n");
        return 0;
    }

    while (fscanf(fp, "%s %s", fileUser, filePass) != EOF) {
        if (strcmp(username, fileUser) == 0 && strcmp(password, filePass) == 0) {
            fclose(fp);
            printf(" Login successful! Welcome, %s.\n", username);
            return 1;
        }
    }
    fclose(fp);
    printf(" Invalid username or password!\n");
    return 0;
}

// ---------------- Stack Functions ----------------
void push(int roomId, char customer[]) {
    struct StackNode* newNode = (struct StackNode*)malloc(sizeof(struct StackNode));
    newNode->roomId = roomId;
    strcpy(newNode->customer, customer);
    newNode->next = top;
    top = newNode;
}

void displayCheckedOut() {
    printf("\nRecently Checked-out Customers (Stack):\n");
    struct StackNode* temp = top;
    if (!temp) {
        printf("No checkouts yet.\n");
        return;
    }
    while (temp) {
        printf("Customer: %s | Room ID: %d\n", temp->customer, temp->roomId);
        temp = temp->next;
    }
}

// ---------------- Queue Functions ----------------
void enqueue(char name[], char roomType[]) {
    struct Node* newNode = (struct Node*)malloc(sizeof(struct Node));
    strcpy(newNode->name, name);
    strcpy(newNode->roomType, roomType);
    newNode->next = NULL;
    if (rear == NULL) {
        front = rear = newNode;
    } else {
        rear->next = newNode;
        rear = newNode;
    }
}

void displayQueue() {
    printf("\nWaiting List (Queue):\n");
    struct Node* temp = front;
    if (!temp) {
        printf("No customers in waiting list.\n");
        return;
    }
    while (temp) {
        printf("%s is waiting for a %s room.\n", temp->name, temp->roomType);
        temp = temp->next;
    }
}

// ---------------- Linked List Functions ----------------
void bookRoom(int roomId, char customer[]) {
    for (int i = 0; i < roomCount; i++) {
        if (hotel[i].id == roomId) {
            if (!hotel[i].isBooked) {
                hotel[i].isBooked = 1;
                struct Booking* newBooking = (struct Booking*)malloc(sizeof(struct Booking));
                newBooking->roomId = roomId;
                strcpy(newBooking->customer, customer);
                newBooking->next = bookingHead;
                bookingHead = newBooking;
                printf("%s booked Room ID %d successfully.\n", customer, roomId);
                return;
            } else {
                printf("Room is already booked. Adding %s to waiting list.\n", customer);
                enqueue(customer, hotel[i].type);
                return;
            }
        }
    }
    printf("Room not found!\n");
}

void checkoutRoom(int roomId) {
    struct Booking *temp = bookingHead, *prev = NULL;
    while (temp) {
        if (temp->roomId == roomId) {
            if (prev) prev->next = temp->next;
            else bookingHead = temp->next;
            hotel[roomId - 1].isBooked = 0;
            push(roomId, temp->customer);
            printf("Customer %s checked out from Room ID %d.\n", temp->customer, roomId);
            free(temp);

            // Assign room to waiting list
            if(front != NULL) {
                char nextCustomer[50];
                strcpy(nextCustomer, front->name);
                printf("Assigning room %d to waiting customer %s.\n", roomId, nextCustomer);
                hotel[roomId - 1].isBooked = 1;
                bookRoom(roomId, nextCustomer);

                struct Node *oldFront = front;
                front = front->next;
                if(front == NULL) rear = NULL;
                free(oldFront);
            }

            return;
        }
        prev = temp;
        temp = temp->next;
    }
    printf("Room not found in booking list!\n");
}

// ---------------- Array Functions ----------------
void addRoom(int id, char type[]) {
    hotel[roomCount].id = id;
    strcpy(hotel[roomCount].type, type);
    hotel[roomCount].isBooked = 0;
    roomCount++;
}

void displayRooms() {
    printf("\nAvailable Rooms (Array):\n");
    for (int i = 0; i < roomCount; i++) {
        printf("ID: %d | Type: %s | Status: %s\n", hotel[i].id, hotel[i].type,
               hotel[i].isBooked ? "Booked" : "Available");
    }
}

// ---------------- Main ----------------
int main() {
    addRoom(1, "Single");
    addRoom(2, "Double");
    addRoom(3, "Suite");

    int mainChoice;
    while (1) {
        printf("\n===== Welcome to Hotel Management System =====\n");
        printf("1. Register\n2. Login\n3. Exit\n");
        printf("Enter choice: ");
        scanf("%d", &mainChoice);

        if (mainChoice == 1) {
            registerUser();
        } else if (mainChoice == 2) {
            if (loginUser()) {
                int choice, id;
                char name[50];
                while (1) {
                    printf("\n===== Hotel Management Menu =====\n");
                    printf("1. Display Rooms\n2. Book Room\n3. Checkout Room\n4. Show Waiting List\n5. Show Checked-out Customers\n6. Logout\n");
                    printf("Enter choice: ");
                    scanf("%d", &choice);

                    switch (choice) {
                        case 1:
                            displayRooms();
                            break;
                        case 2:
                            printf("Enter your name: ");
                            scanf("%s", name);
                            printf("Enter Room ID: ");
                            scanf("%d", &id);
                            bookRoom(id, name);
                            break;
                        case 3:
                            printf("Enter Room ID to checkout: ");
                            scanf("%d", &id);
                            checkoutRoom(id);
                            break;
                        case 4:
                            displayQueue();
                            break;
                        case 5:
                            displayCheckedOut();
                            break;
                        case 6:
                            goto endSession;
                        default:
                            printf("Invalid choice!\n");
                    }
                }
                endSession:;
            }
        } else if (mainChoice == 3) {
            printf("Exiting program. Goodbye!\n");
            break;
        } else {
            printf("Invalid choice!\n");
        }
    }
    return 0;
}
